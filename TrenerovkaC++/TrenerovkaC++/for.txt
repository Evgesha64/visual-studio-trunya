#include <iostream>

int main()
{
    int sum = 0;
    int rengeBegin, rengeEnd;

    setlocale(LC_ALL, "ru");
    std::cout << "Введите значение 1\n";
    std::cin >> rengeBegin;

    std::cout << "Введите значение 2\n";
    std::cin >> rengeEnd;

    std::cout << "Готово\n";

    do
    {
        if (rengeBegin % 2 == 0) {
            sum += rengeBegin;
        }
        rengeBegin++;

    } while (rengeBegin < rengeEnd);

    std::cout << "Всё чётко! " << sum << std::endl;

}
