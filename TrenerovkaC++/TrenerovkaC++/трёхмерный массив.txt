#include <iostream>
using namespace std;

int main()
{
    setlocale(LC_ALL, "ru");

    const int x = 5;
    const int y = 5;
    const int z = 5;



    int arr[x][y][z];

    for (size_t i1 = 0; i1 < x; i1++)
    {

        for (size_t i2 = 0; i2 < y; i2++)
        {

            for (size_t i3 = 0; i3 < z; i3++)
            {
                arr[i1][i2][i3] = i3;
            }

        }

    }

    for (size_t i1 = 0; i1 < x; i1++)
    {
        cout << endl;
        for (size_t i2 = 0; i2 < y; i2++)
        {
            cout << endl;
            for (size_t i3 = 0; i3 < z; i3++)
            {
                cout << " " << arr[i1][i2][i3];
            }
        }

    }


}
